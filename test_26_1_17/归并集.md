#define SIZE 100
int UFSets[SIZE]; // 集合数组

// 1. 初始化
void Initial(int S[]) {
    for (int i = 0; i < SIZE; i++)
        S[i] = -1; // -1 表示自己是根
}

// 2. 查找（含路径压缩优化）—— 考研加分项
int Find(int S[], int x) {
    int root = x;
    while (S[root] >= 0) root = S[root]; // 循环找到根

    // 路径压缩：把沿途所有结点的父节点直接设为根
    while (x != root) {
        int t = S[x];
        S[x] = root;
        x = t;
    }
    return root;
}

// 3. 合并 Union
void Union(int S[], int Root1, int Root2) {
    if (Root1 == Root2) return;
    // 简单合并：将Root2连接到Root1下面
    S[Root2] = Root1;
    // 注意：更高级的优化是“小树并入大树”，但考研通常不考代码实现，只考逻辑
}
